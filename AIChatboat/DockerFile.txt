# ─────────────────────────────────────────────
# Stage 1 — Build the application
# ─────────────────────────────────────────────

# Use official Microsoft .NET Framework SDK image
# This includes all tools needed to BUILD the app
FROM mcr.microsoft.com/dotnet/framework/sdk:4.8 AS build

# Set the working directory inside the container
WORKDIR /app

# Copy all project files into the container
COPY . .

# Restore NuGet packages (downloads Newtonsoft.Json etc.)
RUN nuget restore AIChatbot.sln

# Build the project in Release mode
RUN msbuild AIChatbot.sln /p:Configuration=Release /p:Platform="Any CPU"

# ─────────────────────────────────────────────
# Stage 2 — Runtime image
# ─────────────────────────────────────────────

# Use a smaller runtime-only image to run the app
FROM mcr.microsoft.com/dotnet/framework/runtime:4.8 AS runtime

# Set working directory
WORKDIR /app

# Copy compiled output from build stage
COPY --from=build /app/bin/Release/ .

# Command to run the application
CMD ["AIChatbot.exe"]
```

5. Press **Ctrl + S** to save
```

```
## STEP 3 — Install Docker Desktop

Since Docker needs to be installed to actually run containers:

1. Go to 👉 **https://www.docker.com/products/docker-desktop**
2. Click **"Download for Windows"**
3. Run the installer
4. Restart your PC when asked
5. After restart, Docker Desktop opens — you'll see a **whale icon** in the taskbar

> ✅ Docker Desktop must be **running** (whale icon visible) before you can use docker commands
```
